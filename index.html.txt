  <h1>Register Page with CRUD Operations</h1>
  <form id="registerForm">
    <input type="text" id="name" placeholder="Name" required />
    <input type="email" id="email" placeholder="Email" required />
    <button type="submit">Register</button>
  </form>

  <table>
    <thead>
      <tr>
        <th>Name</th>
        <th>Email</th>
        <th class="center">Actions</th>
      </tr>
    </thead>
    <tbody id="userTableBody">
      <!-- Dynamic Content -->
    </tbody>
  </table>

  <script>
    const form = document.getElementById('registerForm');
    const userTableBody = document.getElementById('userTableBody');

    let users = [];
    let editIndex = -1;

    function renderTable() {
      userTableBody.innerHTML = '';
      users.forEach((user, index) => {
        const tr = document.createElement('tr');

        tr.innerHTML = `
          <td>${user.name}</td>
          <td>${user.email}</td>
          <td class="center">
            <span class="action-icons edit" title="Edit" onclick="editUser(${index})">&#9998;</span>
            <span class="action-icons delete" title="Delete" onclick="deleteUser(${index})">&#10060;</span>
          </td>
        `;

        userTableBody.appendChild(tr);
      });
    }

    function resetForm() {
      form.name.value = '';
      form.email.value = '';
      editIndex = -1;
      form.querySelector('button').textContent = 'Register';
    }

    form.addEventListener('submit', function(e) {
      e.preventDefault();
      const name = form.name.value.trim();
      const email = form.email.value.trim();

      if (!name || !email) return;

      if (editIndex === -1) {
        // Create
        users.push({ name, email });
      } else {
        // Update
        users[editIndex] = { name, email };
      }

      renderTable();
      resetForm();
    });

    function editUser(index) {
      const user = users[index];
      form.name.value = user.name;
      form.email.value = user.email;
      editIndex = index;
      form.querySelector('button').textContent = 'Update';
    }

    function deleteUser(index) {
      if (confirm('Are you sure you want to delete this user?')) {
        users.splice(index, 1);
        renderTable();
        if (editIndex === index) resetForm();
      }
    }

    renderTable();
  </script>
